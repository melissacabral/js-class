// define a variable
let score = 0;
// update the value
score = 10;
//different ways to add
score = score + 10;
score += 10;
//add 1
score ++;
//subtract 1 
score --;
//multiply by 10
score *= 10;

//show the value in the console
console.log(score);

//constants cannot be changed once set
const player = "Mario";
//	redefining a const is not allowed
// player = "luigi";
